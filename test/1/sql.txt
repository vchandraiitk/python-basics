1. Create table with constraints and insert records
CREATE TABLE PROJECT_SUBMISSION
(
    SID INT PRIMARY KEY,
    SNAME VARCHAR(25) NOT NULL,
    MARK INT CHECK (MARK >= 0 AND MARK <= 100),
    STATUS VARCHAR(10) DEFAULT 'PENDING'
);

3. Display students scoring above average and status = 'SUBMITTED'
SELECT *
FROM PROJECT_SUBMISSION
WHERE MARK >
      (SELECT AVG(MARK) FROM PROJECT_SUBMISSION)
AND STATUS = 'SUBMITTED'
ORDER BY MARK DESC;
